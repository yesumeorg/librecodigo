v0.1.0-alpha

SDK para Plugins de Libre Código

Crea plugins que extiendan las capacidades de Libre Código. Registra comandos, paneles, bloques visuales, temas personalizados y mucho más con una API sencilla y documentada.

Empezar API Reference

Instalación

Existen dos formas de integrar el SDK en tu plugin:

Opción 1: Incluir directamente

Añade el archivo del SDK en la carpeta de tu plugin y referéncialo en tu HTML:

HTML
<script src="libre-codigo-sdk.js"></script>

Opción 2: npm (futuro)

bash
npm install @librecodigo/sdk
Nota: La instalación por npm estará disponible en una versión estable futura. Por ahora, incluye el archivo JS directamente en tu proyecto.

Estructura de un Plugin

Cada plugin es una carpeta independiente con la siguiente estructura:

mi-plugin/
├── manifest.json # Metadatos del plugin
├── index.js # Punto de entrada principal
├── styles.css # Estilos opcionales
└── README.md # Documentación del plugin

El archivo manifest.json es obligatorio y define los metadatos esenciales. El index.js es el punto de entrada donde se registra la funcionalidad del plugin usando la API del SDK.

manifest.json

El archivo manifest.json contiene toda la información necesaria para que Libre Código cargue y gestione tu plugin.

JSON
{
  "name": "mi-plugin",
  "displayName": "Mi Plugin",
  "version": "1.0.0",
  "description": "Descripción del plugin",
  "author": "Tu Nombre",
  "main": "index.js",
  "icon": "icon.png",
  "permissions": ["filesystem", "terminal", "editor"],
  "minAppVersion": "0.1.0",
  "dependencies": {},
  "contributes": {
    "commands": [],
    "panels": [],
    "blocks": [],
    "themes": [],
    "languages": []
  }
}

Referencia de la API

Toda la API está disponible a través del objeto global LibreCodigo que el SDK expone tras cargarse.

Sistema de Eventos

El SDK expone un sistema de eventos pub/sub para reaccionar a acciones del IDE. Usa onEvent para escuchar y emitEvent para publicar.

Eventos del Sistema

file-openSe abrió un archivo. Datos: {path, language}
file-saveSe guardó un archivo. Datos: {path}
file-closeSe cerró un archivo. Datos: {path}
code-changeEl contenido del editor cambió. Datos: {changes}
language-changeSe cambió el lenguaje. Datos: {language}
project-changeSe abrió un proyecto. Datos: {projectPath}
project-closeSe cerró el proyecto. Sin datos.
block-addSe añadió un bloque. Datos: {blockId}
block-removeSe eliminó un bloque. Datos: {blockId}
block-changeUn bloque cambió. Datos: {blockId, changes}
terminal-outputSalida del terminal. Datos: {text, stream}

Ejemplo de Uso

JavaScript
const disposable = LibreCodigo.onEvent('file-save', (data) => {
  console.log('Archivo guardado:', data.path);
});

// Más tarde, para desuscribirse:
disposable.dispose();

Eventos Personalizados

Puedes crear y emitir tus propios eventos para la comunicación entre componentes de tu plugin o con otros plugins:

JavaScript
// Emitir un evento personalizado
LibreCodigo.emitEvent('mi-plugin:datorecibido', { id: 42, texto: 'Hola' });

// Escucharlo desde otro plugin
LibreCodigo.onEvent('mi-plugin:datorecibido', (data) => {
  LibreCodigo.showNotification('Recibido: ' + data.texto, 'info');
});

Ejemplo: Plugin de Formateo JSON

Un plugin completo que registra un comando para formatear JSON en el editor.

manifest.json

JSON
{
  "name": "json-formatter",
  "displayName": "JSON Formatter",
  "version": "1.0.0",
  "description": "Formatea y valida JSON en el editor",
  "author": "Libre Código",
  "main": "index.js",
  "permissions": ["editor"],
  "contributes": {
    "commands": [{ "id": "json-format", "label": "Format JSON" }]
  }
}

index.js

JavaScript
LibreCodigo.registerCommand('json-format', {
  label: 'Format JSON',
  category: 'Format',
  callback: () => {
    const editor = LibreCodigo.getEditor();
    const content = editor.getValue();
    try {
      const parsed = JSON.parse(content);
      const formatted = JSON.stringify(parsed, null, 2);
      editor.setValue(formatted);
      LibreCodigo.showNotification('JSON formateado correctamente', 'success');
    } catch (e) {
      LibreCodigo.showNotification('JSON inválido: ' + e.message, 'error');
    }
  }
});

Ejemplo: Plugin de Conteo de Líneas

Un plugin que registra un panel lateral mostrando estadísticas del archivo activo.

manifest.json

JSON
{
  "name": "line-counter",
  "displayName": "Conteo de Líneas",
  "version": "1.0.0",
  "description": "Muestra estadísticas del archivo activo",
  "author": "Libre Código",
  "main": "index.js",
  "permissions": ["editor"],
  "contributes": {
    "panels": [{ "id": "line-stats", "title": "Estadísticas" }]
  }
}

index.js

JavaScript
function getStats() {
  const editor = LibreCodigo.getEditor();
  if (!editor) return { lines: 0, words: 0, chars: 0 };
  const content = editor.getValue();
  return {
    lines: content.split('\n').length,
    words: content.split(/\s+/).filter(Boolean).length,
    chars: content.length
  };
}

LibreCodigo.registerPanel('line-stats', {
  title: 'Estadísticas',
  icon: 'bar-chart-2',
  render: () => {
    const s = getStats();
    return `
      <div style="padding:16px;font-family:Inter,sans-serif">
        <h3 style="margin-bottom:12px">Estadísticas</h3>
        <p>Líneas: <strong>${s.lines}</strong></p>
        <p>Palabras: <strong>${s.words}</strong></p>
        <p>Caracteres: <strong>${s.chars}</strong></p>
      </div>
    `;
  }
});

LibreCodigo.onEvent('code-change', () => {
  LibreCodigo.emitEvent('line-stats:refresh');
});

Publicación

Sigue estos pasos para publicar tu plugin y que la comunidad pueda usarlo:

  1. Crea el manifest.json — Asegúrate de que todos los campos requeridos estén completos y que el name sea único.
  2. Testea localmente — Coloca tu plugin en la carpeta plugins/ del directorio de Libre Código y reinicia la aplicación.
  3. Sube al foro — Crea un hilo en el foro de Libre Código con el código fuente, capturas de pantalla y una descripción.
  4. Recibe feedback — La comunidad probará tu plugin y dejará sugerencias. Itera sobre las mejoras.
  5. Publica v1.0 — Cuando el plugin esté estable, actualiza la versión a 1.0.0 y publica la versión final.
Importante: Todos los plugins pasan por una revisión de seguridad antes de ser incluidos en el repositorio oficial. No incluyas código malicioso, no accedas a permisos innecesarios y documenta tu plugin correctamente.

Descarga del SDK

El SDK completo en un solo archivo JavaScript. Copia, descarga e inclúyelo en tu plugin.

JavaScript — libre-codigo-sdk.js
/**
 * Libre Código Plugin SDK v0.1.0-alpha
 * API completa para la extensión de Libre Código IDE
 * https://librecodigo.org
 *
 * Uso:
 *   <script src="libre-codigo-sdk.js"></script>
 *   // Luego accede a la API via globalThis.LibreCodigo
 */
(function (global) {
  'use strict';

  const VERSION = '0.1.0-alpha';

  /* ─── State ─────────────────────────────────────────────── */

  const _events = Object.create(null);
  const _commands = Object.create(null);
  const _panels = Object.create(null);
  const _blocks = [];
  const _themes = Object.create(null);
  let _idCounter = 0;

  /* ─── Helpers ───────────────────────────────────────────── */

  function _uid() { return '_sdk_' + (++_idCounter); }

  function _on(event, callback) {
    if (!_events[event]) _events[event] = [];
    _events[event].push(callback);
    return {
      dispose: function () {
        const idx = _events[event].indexOf(callback);
        if (idx !== -1) _events[event].splice(idx, 1);
      }
    };
  }

  function _emit(event, data) {
    const listeners = _events[event];
    if (!listeners) return;
    for (let i = 0; i < listeners.length; i++) {
      try { listeners[i](data); } catch (e) { console.error('[SDK] Event error:', e); }
    }
  }

  function _assertStr(v, name) {
    if (typeof v !== 'string' || v.trim() === '') {
      throw new Error('[SDK] ' + name + ' must be a non-empty string');
    }
  }

  function _assertFn(v, name) {
    if (typeof v !== 'function') {
      throw new Error('[SDK] ' + name + ' must be a function');
    }
  }

  /* ─── Mock / Bridge APIs ────────────────────────────────── */

  function _getMonaco() {
    if (global.monaco) return global.monaco;
    if (global.editor && typeof global.editor.getValue === 'function') return global.editor;
    console.warn('[SDK] Monaco editor not found. Returning mock.');
    return {
      getValue: function () { return ''; },
      setValue: function () {},
      getModel: function () {
        return { getValue: function () { return ''; }, setValue: function () {} };
      },
      updateOptions: function () {}
    };
  }

  function _getFs() {
    if (global.LibreCodigo && global.LibreCodigo.__fs) return global.LibreCodigo.__fs;
    console.warn('[SDK] Filesystem bridge not available.');
    return {
      readFile: function () { return Promise.resolve(''); },
      writeFile: function () { return Promise.resolve(); },
      createFile: function () { return Promise.resolve(); },
      deleteFile: function () { return Promise.resolve(); },
      listFiles: function () { return Promise.resolve([]); }
    };
  }

  function _shellExec(cmd) {
    if (global.LibreCodigo && typeof global.LibreCodigo.__exec === 'function') {
      return global.LibreCodigo.__exec(cmd);
    }
    console.warn('[SDK] Shell execution not available.');
    return Promise.resolve({ stdout: '', stderr: 'Not supported', exitCode: 1 });
  }

  /* ─── SDK Public API ────────────────────────────────────── */

  const SDK = {
    version: VERSION,

    /* ── Registration ── */

    registerCommand: function (id, options) {
      _assertStr(id, 'command id');
      if (!options || typeof options !== 'object') {
        throw new Error('[SDK] Command options must be an object');
      }
      _assertStr(options.label, 'command label');
      _assertFn(options.callback, 'command callback');
      _commands[id] = {
        id: id,
        label: options.label,
        category: options.category || 'General',
        callback: options.callback,
        enabled: true
      };
      _emit('command-registered', { id: id });
      return _commands[id];
    },

    registerPanel: function (id, options) {
      _assertStr(id, 'panel id');
      if (!options || typeof options !== 'object') {
        throw new Error('[SDK] Panel options must be an object');
      }
      _assertStr(options.title, 'panel title');
      _assertFn(options.render, 'panel render');
      _panels[id] = {
        id: id,
        title: options.title,
        icon: options.icon || 'layout',
        render: options.render
      };
      _emit('panel-registered', { id: id });
      return _panels[id];
    },

    registerBlock: function (definition) {
      if (!definition || typeof definition !== 'object') {
        throw new Error('[SDK] Block definition must be an object');
      }
      _assertStr(definition.id, 'block id');
      _assertStr(definition.label, 'block label');
      _blocks.push(Object.assign({ category: 'general', shape: 'stack', color: '#C96B4B', ports: [] }, definition));
      _emit('block-registered', { id: definition.id });
    },

    registerTheme: function (name, theme) {
      _assertStr(name, 'theme name');
      if (!theme || typeof theme !== 'object') {
        throw new Error('[SDK] Theme must be an object');
      }
      _themes[name] = theme;
      _emit('theme-registered', { name: name });
    },

    /* ── Editor ── */

    getEditor: function () { return _getMonaco(); },

    getLanguage: function () {
      const m = _getMonaco();
      if (m.getModel && m.getModel()) {
        const lang = m.getModel().getLanguageId ? m.getModel().getLanguageId() : 'unknown';
        return lang;
      }
      return 'plaintext';
    },

    setLanguage: function (lang) {
      _assertStr(lang, 'language');
      const m = _getMonaco();
      if (m.setModelLanguage) m.setModelLanguage(m.getModel(), lang);
      _emit('language-change', { language: lang });
    },

    getTheme: function () {
      const m = _getMonaco();
      if (m.getOption) {
        const t = m.getOption(42);
        return t || 'vs-dark';
      }
      return 'vs-dark';
    },

    getActiveFile: function () {
      if (global.LibreCodigo && global.LibreCodigo.__activeFile) {
        return global.LibreCodigo.__activeFile;
      }
      return null;
    },

    openFile: function (path) {
      _assertStr(path, 'file path');
      _emit('file-open', { path: path });
    },

    /* ── Filesystem ── */

    getFilesystem: function () { return _getFs(); },

    readFile: function (path) {
      _assertStr(path, 'file path');
      return _getFs().readFile(path);
    },

    writeFile: function (path, content) {
      _assertStr(path, 'file path');
      if (typeof content !== 'string') throw new Error('[SDK] content must be a string');
      return _getFs().writeFile(path, content);
    },

    createFile: function (path) {
      _assertStr(path, 'file path');
      return _getFs().createFile(path);
    },

    deleteFile: function (path) {
      _assertStr(path, 'file path');
      return _getFs().deleteFile(path);
    },

    listFiles: function (dir) {
      _assertStr(dir, 'directory path');
      return _getFs().listFiles(dir);
    },

    /* ── Events ── */

    onEvent: function (event, callback) {
      _assertStr(event, 'event name');
      _assertFn(callback, 'event callback');
      return _on(event, callback);
    },

    emitEvent: function (event, data) {
      _assertStr(event, 'event name');
      _emit(event, data);
    },

    /* ── UI ── */

    showNotification: function (message, type) {
      _assertStr(message, 'notification message');
      var validTypes = ['info', 'warning', 'error', 'success'];
      if (type && validTypes.indexOf(type) === -1) {
        console.warn('[SDK] Invalid notification type. Using "info".');
        type = 'info';
      }
      _emit('notification', { message: message, type: type || 'info' });
      if (global.LibreCodigo && typeof global.LibreCodigo.__notify === 'function') {
        global.LibreCodigo.__notify(message, type || 'info');
      } else {
        console.log('[SDK:' + (type || 'info') + '] ' + message);
      }
    },

    showInputBox: function (options) {
      var opts = options || {};
      if (global.LibreCodigo && typeof global.LibreCodigo.__inputBox === 'function') {
        return global.LibreCodigo.__inputBox(opts);
      }
      var result = prompt(opts.prompt || 'Input:', opts.default || '');
      return Promise.resolve(result);
    },

    showQuickPick: function (items) {
      if (!Array.isArray(items) || items.length === 0) {
        return Promise.resolve(null);
      }
      if (global.LibreCodigo && typeof global.LibreCodigo.__quickPick === 'function') {
        return global.LibreCodigo.__quickPick(items);
      }
      var labels = items.map(function (item) {
        return typeof item === 'string' ? item : (item.label || item);
      });
      var choice = prompt(
        'Choose:\n' + labels.map(function (l, i) { return (i + 1) + '. ' + l; }).join('\n') +
        '\n\nEnter number:'
      );
      var idx = parseInt(choice, 10) - 1;
      return Promise.resolve(idx >= 0 && idx < items.length ? labels[idx] : null);
    },

    getConfiguration: function () {
      if (global.LibreCodigo && global.LibreCodigo.__config) {
        return Object.assign({}, global.LibreCodigo.__config);
      }
      return { theme: 'vs-dark', fontSize: 14, tabSize: 2 };
    },

    /* ── System ── */

    executeCommand: function (cmd) {
      _assertStr(cmd, 'command');
      return _shellExec(cmd);
    },

    /* ── Internals ── */

    _commands: _commands,
    _panels: _panels,
    _blocks: _blocks,
    _themes: _themes,

    getRegisteredCommands: function () {
      return Object.keys(_commands).map(function (k) { return _commands[k]; });
    },

    getRegisteredPanels: function () {
      return Object.keys(_panels).map(function (k) { return _panels[k]; });
    },

    getRegisteredBlocks: function () {
      return _blocks.slice();
    },

    getRegisteredThemes: function () {
      return Object.keys(_themes).map(function (k) { return { name: k, theme: _themes[k] }; });
    },

    unregisterCommand: function (id) {
      if (_commands[id]) { delete _commands[id]; _emit('command-unregistered', { id: id }); }
    },

    unregisterPanel: function (id) {
      if (_panels[id]) { delete _panels[id]; _emit('panel-unregistered', { id: id }); }
    }
  };

  /* ─── Bootstrap ─────────────────────────────────────────── */

  global.LibreCodigo = SDK;

  _emit('sdk-loaded', { version: VERSION });

})(typeof window !== 'undefined' ? window : typeof globalThis !== 'undefined' ? globalThis : this);

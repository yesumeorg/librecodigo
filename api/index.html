BlockCodigo API

Motor de Programación Visual para Libre Código. Transforma código en bloques y bloques en código para más de 21 lenguajes de programación.

Empezar API Reference

Conceptos Fundamentales

BlockCodigo se compone de cuatro pilares que trabajan juntos para lograr la programación visual multilenguaje.

IR (Intermediate Representation)

AST universal intermedia. Todos los lenguajes se parsean hacia esta representación común, permitiendo la conversión entre cualquier par de lenguajes soportados.

Bloques

Representación visual de operaciones de código. Cada bloque tiene una forma, color, puertos de datos y lógica de conversión a/desde IR.

Converters

Transformadores bidireccionales entre IR y Bloques. Convierten AST en bloques arrastrables y reconstruyen AST a partir de la composición de bloques.

Emitters

Generadores de código fuente desde IR para más de 21 lenguajes. Cada emisor conoce la sintaxis y convenciones de su lenguaje destino.

Tipos de Bloques

Cada tipo de bloque tiene una forma visual distinta que representa su función en el programa.

Cuando el programa empiece

Hat (Evento)

Forma redondeada superior, fondo ámbar. Representa un evento que inicia la ejecución. Solo puede ir al inicio de una pila de bloques.

Imprimir en consola "Hola"

Stack (Sentencia)

Rectángulo con borde terracota. Representa una instrucción o sentencia. Se apilan verticalmente formando la secuencia de ejecución.

Variable x

Reporter (Expresión)

Forma de pastilla redondeada, fondo arena. Retorna un valor. Se inserta en los puertos de entrada de otros bloques.

x > 10

Boolean

Forma hexagonal, fondo verde. Retorna verdadero o falso. Se usa en condiciones de bloques de control.

Si / Entonces
arrastra bloques aquí

C-Block (Control)

Forma de C, fondo púrpura. Envuelve otros bloques. Ejemplos: Si/Entonces, Para cada, Mientras, Repetir.

API de Registro de Bloques

Usa la API de BlockCodigo para definir y registrar bloques personalizados que se integran con el motor de programación visual.

Definición de un Bloque

JavaScript
BlockCodigo.registerBlock({
  id: 'print',
  label: 'Imprimir en consola',
  category: 'io',
  shape: 'stack',
  color: '#C96B4B',
  ports: [
    {
      id: 'message',
      label: 'mensaje',
      type: 'string',
      direction: 'input'
    }
  ],
  toCode: (block) => {
    const msg = block.getPortValue('message');
    return `console.log(${msg});`;
  },
  fromCode: (node) => {
    // Parsear nodo AST de vuelta a bloque
    return { portValues: { message: node.arguments[0] } };
  }
});

Pipeline de Conversión

BlockCodigo opera en dos direcciones: de código a bloques (para visualización) y de bloques a código (para ejecución).

Nota: El IR (Intermediate Representation) es el punto central del pipeline. Al usar una representación intermedia universal, BlockCodigo puede convertir entre cualquier par de lenguajes sin necesidad de N×M traductores, solo N parsers + M emitters.

Descarga de BlockCodigo

El motor completo de programación visual en un solo archivo JavaScript. Incluye registro de bloques, conversión IR, emisión multi-lenguaje y renderizado de elementos DOM.

JavaScript — block-codigo.js
/**
 * BlockCodigo — Visual Block Programming API
 * v0.1.0-alpha
 * Motor de programación visual para Libre Código
 *
 * Uso:
 *   <script src="block-codigo.js"></script>
 *   // Accede via globalThis.BlockCodigo
 */
const BlockCodigo = (function () {
  'use strict';

  /* ─── State ────────────────────────────────────────────── */

  const _blocks = new Map();
  const _categories = new Map();
  const _listeners = Object.create(null);
  let _currentLanguage = 'javascript';
  let _idCounter = 0;

  /* ─── Default Categories ───────────────────────────────── */

  const DEFAULT_CATEGORIES = [
    { id: 'program',    label: 'Programa',      color: '#D97B2E', icon: 'play' },
    { id: 'variables',  label: 'Variables',      color: '#9A9088', icon: 'box' },
    { id: 'control',    label: 'Control',        color: '#C96B4B', icon: 'git-branch' },
    { id: 'io',         label: 'Entrada/Salida', color: '#874418', icon: 'terminal' },
    { id: 'math',       label: 'Matemáticas',    color: '#B85C24', icon: 'calculator' },
    { id: 'strings',    label: 'Cadenas',        color: '#D97B2E', icon: 'type' },
    { id: 'operators',  label: 'Operadores',     color: '#C96B4B', icon: 'equal' },
    { id: 'arrays',     label: 'Arreglos',       color: '#874418', icon: 'list' },
    { id: 'functions',  label: 'Funciones',      color: '#D97B2E', icon: 'lambda' },
    { id: 'classes',    label: 'Clases',         color: '#C96B2C', icon: 'box' }
  ];

  DEFAULT_CATEGORIES.forEach(function (cat) {
    _categories.set(cat.id, cat);
  });

  /* ─── Supported Languages ──────────────────────────────── */

  const SUPPORTED_LANGUAGES = [
    'javascript', 'python', 'typescript', 'java', 'csharp', 'cpp',
    'c', 'ruby', 'go', 'rust', 'php', 'swift', 'kotlin',
    'dart', 'r', 'lua', 'haskell', 'elixir', 'scala', 'julia', 'pseudocode'
  ];

  /* ─── Language Emitter Maps ─────────────────────────────── */

  const _emitters = {
    javascript: {
      print: function (msg) { return 'console.log(' + msg + ');'; },
      variable: function (name, val) { return 'let ' + name + ' = ' + val + ';'; },
      assign: function (name, val) { return name + ' = ' + val + ';'; },
      if: function (cond, body) { return 'if (' + cond + ') {\n' + _indent(body) + '\n}'; },
      forLoop: function (varName, from, to, body) {
        return 'for (let ' + varName + ' = ' + from + '; ' + varName + ' < ' + to + '; ' + varName + '++) {\n' + _indent(body) + '\n}';
      },
      whileLoop: function (cond, body) { return 'while (' + cond + ') {\n' + _indent(body) + '\n}'; },
      functionDef: function (name, params, body) {
        return 'function ' + name + '(' + params.join(', ') + ') {\n' + _indent(body) + '\n}';
      },
      functionCall: function (name, args) { return name + '(' + args.join(', ') + ');'; },
      returnVal: function (val) { return 'return ' + val + ';'; },
      arithmetic: function (op, left, right) { return '(' + left + ' ' + op + ' ' + right + ')'; },
      comparison: function (op, left, right) { return '(' + left + ' ' + op + ' ' + right + ')'; },
      classDef: function (name, body) { return 'class ' + name + ' {\n' + _indent(body) + '\n}'; },
      arrayCreate: function (items) { return '[' + items.join(', ') + ']'; },
      arrayGet: function (arr, idx) { return arr + '[' + idx + ']'; },
      stringConcat: function (parts) { return parts.join(' + '); },
      stringLength: function (str) { return str + '.length'; }
    },
    python: {
      print: function (msg) { return 'print(' + msg + ')'; },
      variable: function (name, val) { return name + ' = ' + val; },
      assign: function (name, val) { return name + ' = ' + val; },
      if: function (cond, body) { return 'if ' + cond + ':\n' + _indent(body); },
      forLoop: function (varName, from, to, body) {
        return 'for ' + varName + ' in range(' + from + ', ' + to + '):\n' + _indent(body);
      },
      whileLoop: function (cond, body) { return 'while ' + cond + ':\n' + _indent(body); },
      functionDef: function (name, params, body) {
        return 'def ' + name + '(' + params.join(', ') + '):\n' + _indent(body);
      },
      functionCall: function (name, args) { return name + '(' + args.join(', ') + ')'; },
      returnVal: function (val) { return 'return ' + val; },
      arithmetic: function (op, left, right) { return '(' + left + ' ' + op + ' ' + right + ')'; },
      comparison: function (op, left, right) { return left + ' ' + op + ' ' + right; },
      classDef: function (name, body) { return 'class ' + name + ':\n' + _indent(body); },
      arrayCreate: function (items) { return '[' + items.join(', ') + ']'; },
      arrayGet: function (arr, idx) { return arr + '[' + idx + ']'; },
      stringConcat: function (parts) { return ' + '.join(parts); },
      stringLength: function (str) { return 'len(' + str + ')'; }
    },
    typescript: {
      print: function (msg) { return 'console.log(' + msg + ');'; },
      variable: function (name, val) { return 'const ' + name + ' = ' + val + ';'; },
      assign: function (name, val) { return name + ' = ' + val + ';'; },
      if: function (cond, body) { return 'if (' + cond + ') {\n' + _indent(body) + '\n}'; },
      forLoop: function (varName, from, to, body) {
        return 'for (let ' + varName + ': number = ' + from + '; ' + varName + ' < ' + to + '; ' + varName + '++) {\n' + _indent(body) + '\n}';
      },
      whileLoop: function (cond, body) { return 'while (' + cond + ') {\n' + _indent(body) + '\n}'; },
      functionDef: function (name, params, body) {
        return 'function ' + name + '(' + params.join(', ') + '): void {\n' + _indent(body) + '\n}';
      },
      functionCall: function (name, args) { return name + '(' + args.join(', ') + ');'; },
      returnVal: function (val) { return 'return ' + val + ';'; },
      arithmetic: function (op, left, right) { return '(' + left + ' ' + op + ' ' + right + ')'; },
      comparison: function (op, left, right) { return '(' + left + ' ' + op + ' ' + right + ')'; },
      classDef: function (name, body) { return 'class ' + name + ' {\n' + _indent(body) + '\n}'; },
      arrayCreate: function (items) { return '[' + items.join(', ') + ']'; },
      arrayGet: function (arr, idx) { return arr + '[' + idx + ']'; },
      stringConcat: function (parts) { return parts.join(' + '); },
      stringLength: function (str) { return str + '.length'; }
    }
  };

  function _indent(code) {
    return code.split('\n').map(function (line) { return '  ' + line; }).join('\n');
  }

  /* ─── Events ───────────────────────────────────────────── */

  function _on(event, callback) {
    if (!_listeners[event]) _listeners[event] = [];
    _listeners[event].push(callback);
    return {
      dispose: function () {
        var idx = _listeners[event].indexOf(callback);
        if (idx !== -1) _listeners[event].splice(idx, 1);
      }
    };
  }

  function _emit(event, data) {
    var list = _listeners[event];
    if (!list) return;
    for (var i = 0; i < list.length; i++) {
      try { list[i](data); } catch (e) { console.error('[BlockCodigo] Event error:', e); }
    }
  }

  /* ─── API ──────────────────────────────────────────────── */

  return {
    version: '0.1.0-alpha',

    /* ── Registration ── */

    registerBlock: function (def) {
      if (!def || typeof def !== 'object') throw new Error('[BlockCodigo] Block definition must be an object');
      if (!def.id || typeof def.id !== 'string') throw new Error('[BlockCodigo] Block id must be a string');
      if (!def.label || typeof def.label !== 'string') throw new Error('[BlockCodigo] Block label must be a string');
      if (!def.category || !_categories.has(def.category)) {
        throw new Error('[BlockCodigo] Invalid category: ' + def.category);
      }
      if (['hat', 'stack', 'reporter', 'boolean', 'c-block'].indexOf(def.shape) === -1) {
        throw new Error('[BlockCodigo] Invalid shape: ' + def.shape);
      }
      var block = Object.assign({
        ports: [],
        tooltip: '',
        toCode: function () { return ''; },
        fromCode: function () { return {}; }
      }, def);
      _blocks.set(block.id, block);
      _emit('block-registered', { id: block.id, block: block });
      return block;
    },

    registerCategory: function (cat) {
      if (!cat || !cat.id || !cat.label || !cat.color) {
        throw new Error('[BlockCodigo] Category requires id, label, and color');
      }
      _categories.set(cat.id, cat);
      _emit('category-registered', { id: cat.id });
    },

    getBlock: function (id) {
      return _blocks.has(id) ? Object.assign({}, _blocks.get(id)) : null;
    },

    getAllBlocks: function () {
      var result = [];
      _blocks.forEach(function (block) { result.push(Object.assign({}, block)); });
      return result;
    },

    getBlocksByCategory: function (catId) {
      var result = [];
      _blocks.forEach(function (block) {
        if (block.category === catId) result.push(Object.assign({}, block));
      });
      return result;
    },

    getCategories: function () {
      var result = [];
      _categories.forEach(function (cat) { result.push(Object.assign({}, cat)); });
      return result;
    },

    /* ── Language ── */

    getLanguage: function () { return _currentLanguage; },

    setLanguage: function (lang) {
      if (SUPPORTED_LANGUAGES.indexOf(lang) === -1) {
        throw new Error('[BlockCodigo] Unsupported language: ' + lang + '. Supported: ' + SUPPORTED_LANGUAGES.join(', '));
      }
      var prev = _currentLanguage;
      _currentLanguage = lang;
      _emit('language-changed', { language: lang, previous: prev });
    },

    getSupportedLanguages: function () { return SUPPORTED_LANGUAGES.slice(); },

    /* ── Conversion ── */

    convertCodeToBlocks: function (code, language) {
      var lang = language || _currentLanguage;
      if (typeof code !== 'string') throw new Error('[BlockCodigo] Code must be a string');
      var blocks = [];
      var lines = code.split('\n');
      for (var i = 0; i < lines.length; i++) {
        var line = lines[i].trim();
        if (line === '') continue;
        var detected = _detectBlockFromLine(line);
        if (detected) blocks.push(detected);
      }
      _emit('code-converted', { code: code, language: lang, blocks: blocks });
      return blocks;
    },

    convertBlocksToCode: function (blocks, language) {
      var lang = language || _currentLanguage;
      if (!Array.isArray(blocks)) throw new Error('[BlockCodigo] blocks must be an array');
      var emitter = _emitters[lang] || _emitters['javascript'];
      var lines = [];
      for (var i = 0; i < blocks.length; i++) {
        var block = blocks[i];
        var def = _blocks.get(block.blockId || block.id);
        if (def && typeof def.toCode === 'function') {
          var codeLine = def.toCode(block);
          if (codeLine) lines.push(codeLine);
        }
      }
      var result = lines.join('\n');
      _emit('blocks-converted', { blocks: blocks, language: lang, code: result });
      return result;
    },

    /* ── DOM Rendering ── */

    createBlockElement: function (blockDef) {
      if (!blockDef || !blockDef.shape) return null;
      var el = document.createElement('div');
      el.className = 'bc-block bc-block--' + blockDef.shape;
      el.dataset.blockId = blockDef.id;
      el.style.cssText = 'display:inline-flex;align-items:center;gap:8px;padding:10px 18px;font-family:Inter,sans-serif;font-size:14px;font-weight:500;border-radius:6px;cursor:grab;user-select:none;';

      switch (blockDef.shape) {
        case 'hat':
          el.style.background = blockDef.color || '#F0B429';
          el.style.color = '#222';
          el.style.borderRadius = '20px 20px 6px 6px';
          break;
        case 'stack':
          el.style.border = '3px solid ' + (blockDef.color || '#C96B4B');
          el.style.background = (blockDef.color || '#C96B4B') + '10';
          el.style.color = '#222';
          break;
        case 'reporter':
          el.style.background = '#F5E6C8';
          el.style.border = '2px solid #c4a97d';
          el.style.color = '#222';
          el.style.borderRadius = '20px';
          break;
        case 'boolean':
          el.style.background = '#3DA34D';
          el.style.color = '#fff';
          el.style.clipPath = 'polygon(12px 0, calc(100% - 12px) 0, 100% 50%, calc(100% - 12px) 100%, 12px 100%, 0 50%)';
          el.style.padding = '10px 24px';
          break;
        case 'c-block':
          el.style.border = '3px solid ' + (blockDef.color || '#7B61FF');
          el.style.background = (blockDef.color || '#7B61FF') + '10';
          el.style.borderRadius = '8px';
          el.style.flexDirection = 'column';
          el.style.alignItems = 'stretch';
          break;
      }

      var label = document.createElement('span');
      label.textContent = blockDef.label;
      el.appendChild(label);

      if (blockDef.ports) {
        blockDef.ports.forEach(function (port) {
          var portEl = document.createElement('span');
          portEl.className = 'bc-port';
          portEl.dataset.portId = port.id;
          portEl.textContent = port.label || port.id;
          portEl.style.cssText = 'background:#fff;border:2px solid ' + (blockDef.color || '#C96B4B') + ';border-radius:4px;padding:4px 12px;font-family:"JetBrains Mono",monospace;font-size:12px;min-width:60px;';
          el.appendChild(portEl);
        });
      }

      _emit('block-rendered', { id: blockDef.id, element: el });
      return el;
    },

    /* ── Events ── */

    on: function (event, callback) { return _on(event, callback); },

    emit: function (event, data) { _emit(event, data); }
  };

  /* ─── Helpers (private) ─────────────────────────────────── */

  function _detectBlockFromLine(line) {
    var result = { id: _uid(), ports: {} };

    if (/console\.log|print\(|puts |println|System\.out/.test(line)) {
      var msgMatch = line.match(/(?:console\.log|print|puts|println|System\.out\.println)\s*\(\s*(.+?)\s*\)/);
      result.blockId = 'print';
      if (msgMatch) result.ports.message = msgMatch[1];
      return result;
    }
    if (/^for\s*\(/.test(line) || /^for\s+\w+\s+in\s+range/.test(line)) {
      result.blockId = 'forLoop';
      return result;
    }
    if (/^while\s*[\(]/.test(line)) {
      result.blockId = 'whileLoop';
      return result;
    }
    if (/^if\s*[\(]/.test(line) || /^if\s+\w/.test(line)) {
      result.blockId = 'ifCondition';
      return result;
    }
    if (/^(let|const|var|int|string|float)\s+\w+\s*=/.test(line)) {
      result.blockId = 'variable';
      var varMatch = line.match(/(?:let|const|var|int|string|float)\s+(\w+)\s*=\s*(.+)/);
      if (varMatch) { result.ports.name = varMatch[1]; result.ports.value = varMatch[2]; }
      return result;
    }
    if (/function\s+\w+|def\s+\w+/.test(line)) {
      result.blockId = 'functionDef';
      return result;
    }
    if (/return\s+/.test(line)) {
      result.blockId = 'returnVal';
      var retMatch = line.match(/return\s+(.+)/);
      if (retMatch) result.ports.value = retMatch[1];
      return result;
    }
    if (/class\s+\w+/.test(line)) {
      result.blockId = 'classDef';
      return result;
    }
    result.blockId = 'unknown';
    result.ports.code = line;
    return result;
  }

  function _uid() { return 'bc_' + (++_idCounter); }

})();

/* Export for module environments */
if (typeof module !== 'undefined' && module.exports) {
